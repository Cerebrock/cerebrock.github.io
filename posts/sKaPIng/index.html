# removing duplicates from dictionary with fuzzy string matching

seen = []
kpis2 = defaultdict(list)

for k in d:
    for v in d[k]:
        pres = False
        for s in seen:
            if fuzz.token_sort_ratio(s.lower(), v.lower()) > 85:    
                #print(f"{v}, {s}")
                pres = True
                break
        if not pres:
            kpis2[k].append(v)
            seen.append(v)
